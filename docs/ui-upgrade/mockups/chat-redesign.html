import pandas as pd

df = pd.read_csv(DATA / "fifa_world_cup_2026_player_performance.csv")
df = df[df.minutes > 0]                     # drop bench/DNP rows (42%)

# per-player rating: performance blended with market signal
agg = df.groupby(["team", "player_name", "position"]).agg(
    quality=("performance_score", "mean"),
    minutes=("minutes", "sum"),
    age=("age", "first"),
    value=("market_value_eur_m", "first"),
).reset_index()

# shrink low-minute players toward position mean
pos_mean = agg.groupby("position").quality.transform("mean")
w = (agg.minutes / 450).clip(upper=1.0)
agg["rating"] = w * agg.quality + (1 - w) * pos_mean
